Creating spies

fn = sinon.spy()
fn()
fn.calledOnce == true
fn.callCount == 1

Spying/stubbing

sinon.spy($, 'ajax')
$.ajax();
$.ajax.calledOnce == true
sinon.stub($, 'ajax', function () { ... }) // function optional
$.ajax.calledWithMatch({ url: '/x' })
$.ajax.restore()

Anonymous stub

stub = sinon.stub().returns(42)
stub() == 42
stub
  .withArgs(42).returns(1)
  .withArgs(43).throws("TypeError")
stub
  .returns(1)
  .throws("TypeError")
  .returnsArg(0) // Return 1st argument
  .callsArg(0)
